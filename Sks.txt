create a new repository on the command line.

echo "# test" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/sushantkr961/test.git
git push -u origin main

git remote add origin https://github.com/sushantkr961/test.git
git branch -M main
git push -u origin main
